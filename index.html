Weather Notifier
Using Python

This program fetches the live weather information from a specific URL using web scraping, displays the current temperature and weather condition for a given location using Windows 10 notifications.

Weather Notifier

This program imports necessary libraries, retrieves weather data from a specific website, parses the HTML content using BeautifulSoup, extracts current temperature, temperature type, and region information, and displays a notification with the live weather update on Windows using the win10toast library.


from bs4 import BeautifulSoup
from win10toast import ToastNotifier
import urllib.request                                           
n = ToastNotifier()                                                            
response = urllib.request.urlopen(
    "https://weather.com/en-IN/weather/today/l/dc224dd10da9029e8170833dc8f204c65eb430181c1d2f014c1c7daeb5b4b3a5"
)
hmldata = response.read()
soup = BeautifulSoup(hmldata, "html.parser")
current_temp = soup.find_all("span", class_="CurrentConditions--tempValue--MHmYY")
temp_type = soup.find_all("div", class_="CurrentConditions--phraseValue--mZC_p")
region = soup.find_all("h1", class_="CurrentConditions--location--1YWj_")
temp = str(current_temp)
temp_t = str(temp_type)
loc = str(region)
result = (
    "Current Temperature " + temp[82:-8] + "  in " + loc[48:-6] + "\n" + temp_t[75:-7]
)
n.show_toast("Live Weather update", result, duration=10)

Using Python

This program is a graphical user interface (GUI) application implemented using the Tkinter library in Python. It allows the user to play the game "Rock Paper Scissors" against the computer.

Rock-Paper-Scissors

This program is a graphical user interface (GUI) application implemented using the Tkinter library in Python. It allows the user to play the game "Rock Paper Scissors" against the computer. The program generates random numbers representing the computer's choices, displays the corresponding images for the choices, determines the winner based on the game rules, and keeps track of the score for both the player and the computer.


    from tkinter import *
    import random
    from PIL import Image, ImageTk
    
    
    def numbergen():
        y = random.randint(1, 3)
        return y
    
    
    def winner(n1, n2):
        if n1 != n2:
            if n1 == 1:
                if n2 == 2:
                    return "pla"
                else:
                    return "com"
            elif n1 == 2:
                if n2 == 1:
                    return "com"
                else:
                    return "pla"
            else:
                if n2 == 1:
                    return "pla"
                else:
                    return "com"
        else:
            return "none"
    
    
    def imggen(n):
        if n == 1:
            return "F:/images/rockimagehd.jpg"
        elif n == 2:
            return "F:/images/paperimghd.jpg"
        else:
            return "F:/images/secimagehd.jpg"
    
    
    m = Tk()
    m.title("ROCK PAPER SCISSORS")
    m.geometry("1190x600")
    m.resizable(False, False)
    m.config(bg="#1CCAD8")
    com_wins = 0
    com_score = Label(
        m,
        text="Computer wins=" + str(com_wins),
        bg="#1CCAD8",
        font=("Helvetica", 20),
    )
    com_score.place(x=0, y=0)
    
    ti_path = "C:/Users/hp/OneDrive/Desktop/bg.jpg"
    ti = Image.open(ti_path)
    ti = ti.resize((170, 160))
    ti = ImageTk.PhotoImage(ti)
    
    comans = Label(m, image=ti, bg="#1CCAD8", borderwidth=3, relief="groove")
    comans.place(x=340, y=100)
    
    pla_wins = 0
    pla_score = Label(
        m,
        text="Player wins=" + str(pla_wins),
        bg="#1CCAD8",
        font=("Helvetica", 20),
    )
    pla_score.place(x=1000, y=0)
    
    plaans = Label(m, image=ti, bg="#1CCAD8", borderwidth=3, relief="groove")
    plaans.place(x=680, y=100)
    
    
    def button_click(pa):
        global com_wins
        global pla_wins
        ca = numbergen()
    
        ci_path = imggen(ca)
        ci = Image.open(ci_path)
        ci = ci.resize((170, 160))
        ci = ImageTk.PhotoImage(ci)
        comans.config(image=ci)
        comans.image = ci
    
        pi_path = imggen(pa)
        pi = Image.open(pi_path)
        pi = pi.resize((170, 160))
        pi = ImageTk.PhotoImage(pi)
        plaans.config(image=pi)
        plaans.image = pi
    
        if winner(ca, pa) == "pla":
            pla_wins += 1
            pla_score.config(text="Player wins=" + str(pla_wins))
        elif winner(ca, pa) == "com":
            com_wins += 1
            com_score.config(text="Computer wins=" + str(com_wins))
    
    
    ri_path = "F:/images/rockimagehd.jpg"
    ri = Image.open(ri_path)
    ri = ri.resize((170, 160))
    rp = ImageTk.PhotoImage(ri)
    rock = Button(m, image=rp, bg="#1CCAD8", relief="flat", command=lambda: button_click(1))
    rock.place(x=170, y=350)
    
    pi_path = "F:/images/paperimghd.jpg"
    pi = Image.open(pi_path)
    pi = pi.resize((170, 160))
    pp = ImageTk.PhotoImage(pi)
    paper = Button(
        m, image=pp, bg="#1CCAD8", relief="flat", command=lambda: button_click(2)
    )
    paper.place(x=510, y=350)
    
    si_path = "F:/images/secimagehd.jpg"
    si = Image.open(si_path)
    si = si.resize((170, 160))
    sp = ImageTk.PhotoImage(si)
    scissors = Button(
        m, image=sp, bg="#1CCAD8", relief="flat", command=lambda: button_click(3)
    )
    scissors.place(x=850, y=350)
    m.mainloop()
